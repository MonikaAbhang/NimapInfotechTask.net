create procedure [dbo].[UpdateCustomer]
(
	@CustomerId int,
		@CustomerName varchar(10)
)
as
begin UPDATE Customers
SET
 CustomerId=@CustomerId,CustomerName=@CustomerName,
		where CustomerId = @CustomerId
end
